486593#include #include // Thư viện cho hàm sqrt using namespace std; // Hàm kiểm tra số chính phương bool isPerfectSquare(int n) { int sqrtN = sqrt(n); return (sqrtN * sqrtN == n); } // Hàm kiểm tra số nguyên tố bool isPrime(int n) { if (n < 2) return false; int i = 2; while (i <= sqrt(n)) { if (n % i == 0) return false; i++; } return true; } // Hàm kiểm tra số đối xứng bool isPalindrome(int n) { int reversed = 0, original = n; while (n > 0) { reversed = reversed * 10 + n % 10; n /= 10; } return (original == reversed); } // Hàm tìm chữ số lớn nhất và nhỏ nhất void findMaxMinDigits(int n, int &maxDigit, int &minDigit) { maxDigit = 0; minDigit = 9; while (n > 0) { int digit = n % 10; if (digit > maxDigit) maxDigit = digit; if (digit < minDigit) minDigit = digit; n /= 10; } } // Hàm kiểm tra các chữ số có tăng dần hay giảm dần không bool isIncreasing(int n) { int prevDigit = 10; // Lớn hơn bất kỳ chữ số nào while (n > 0) { int currentDigit = n % 10; if (currentDigit >= prevDigit) return false; prevDigit = currentDigit; n /= 10; } return true; } bool isDecreasing(int n) { int prevDigit = -1; // Nhỏ hơn bất kỳ chữ số nào while (n > 0) { int currentDigit = n % 10; if (currentDigit <= prevDigit) return false; prevDigit = currentDigit; n /= 10; } return true; } int main() { int n; cout << "Nhap so nguyen duong n: "; cin >> n; // Kiểm tra số chính phương if (isPerfectSquare(n)) { cout << n << " la so chinh phuong." << endl; } else { cout << n << " khong phai la so chinh phuong." << endl; } // Kiểm tra số nguyên tố if (isPrime(n)) { cout << n << " la so nguyen to." << endl; } else { cout << n << " khong phai la so nguyen to." << endl; } // Kiểm tra số đối xứng if (isPalindrome(n)) { cout << n << " la so doi xung." << endl; } else { cout << n << " khong phai la so doi xung." << endl; } // Tìm chữ số lớn nhất và nhỏ nhất int maxDigit, minDigit; findMaxMinDigits(n, maxDigit, minDigit); cout << "Chu so lon nhat: " << maxDigit << endl; cout << "Chu so nho nhat: " << minDigit << endl; // Kiểm tra dãy chữ số có tăng dần hay giảm dần if (isIncreasing(n)) { cout << "Cac chu so cua " << n << " tang dan." << endl; } else if (isDecreasing(n)) { cout << "Cac chu so cua " << n << " giam dan." << endl; } else { cout << "Cac chu so cua " << n << " khong tang dan va khong giam dan." << endl; } return 0; }
